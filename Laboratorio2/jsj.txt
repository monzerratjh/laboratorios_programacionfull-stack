<script>
    // Seleccionamos el formulario con ID 'formTabla' y le agregamos un event listener
    document.getElementById("formTabla").addEventListener("submit", function(e) {
        // â›” Previene que el formulario se envÃ­e de la forma tradicional (recargando la pÃ¡gina)
        e.preventDefault();

        // ðŸ“¦ Crea un objeto FormData con todos los datos del formulario
        // Esto permite enviar los datos como si se hubiera hecho un formulario "normal"
        let formData = new FormData(this);

        // ðŸ“¡ Usamos la funciÃ³n fetch() para enviar los datos al servidor
        // fetch() permite hacer solicitudes HTTP (como POST o GET) sin recargar la pÃ¡gina
        fetch("parte1.php", { // URL del archivo PHP que procesarÃ¡ los datos
            method: "POST",     // MÃ©todo de envÃ­o: POST (porque queremos enviar datos)
            body: formData      // El contenido que estamos enviando al servidor (en este caso, el nÃºmero de la tabla)
        })
        .then(res => res.text()) // ðŸ” Cuando recibimos la respuesta del servidor, la convertimos a texto plano
        .then(data => {
            // ðŸ§¾ El contenido que devuelve el servidor (las tablas) se muestra dentro del div con ID 'resultado'
            // Con innerHTML insertamos ese contenido directamente en el HTML
            document.getElementById("resultado").innerHTML = data;
        });
    });
</script>
